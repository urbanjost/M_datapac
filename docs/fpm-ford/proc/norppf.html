99002    FORMAT (' ','***** THE VALUE OF THE ARGUMENT IS ',E15.8,       &

         RETURN
!
!-----START POINT-----------------------------------------------------
!
      ELSEIF ( P/=0.5 ) THEN
!
         r = P
         IF ( P>0.5 ) r = 1.0 - r
         t = SQRT(-2.0*ALOG(r))
         anum = ((((t*p4+p3)*t+p2)*t+p1)*t+p0)
         aden = ((((t*q4+q3)*t+q2)*t+q1)*t+q0)
         Ppf = t + (anum/aden)

      ENDIF

      RETURN
