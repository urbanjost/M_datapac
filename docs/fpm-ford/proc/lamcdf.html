         ENDIF
      ENDIF

      ENDIF

!

            pupper = pmax
            icount = 0
            DO
               xcalc = (pmid**Alamba-(1.0-pmid)**Alamba)/Alamba
               IF ( xcalc==X ) THEN
                  Cdf = pmid
                  GOTO 99999
               ELSE

                     pupper = pmid

                  ELSE

                  ENDIF
               ENDIF
            ENDDO
         ENDIF
      ENDIF

         RETURN
      ELSE
         Cdf = EXP(X)/(1.0+EXP(X))
